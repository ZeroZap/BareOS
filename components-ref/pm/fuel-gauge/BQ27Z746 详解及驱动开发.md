## 一、BQ27Z746 概述

BQ27Z746 是德州仪器 (TI) 推出的一款高度集成的单芯锂离子/锂聚合物电池电量监测计和保护解决方案,采用专利的 **Impedance Track™** 技术。

### 主要特性

|   特性类别   |           详细说明            |
| :----------: | :---------------------------: |
|  **处理器**  | 闪存可编程定制 BQBMP RISC CPU |
| **通信接口** |    400kHz I2C 总线通信接口    |
| **安全认证** |   SHA-256 安全散列算法认证    |
| **工作电压** |      低电压 (2.0V) 运行       |
|   **ADC**    |     2个16位独立高精度ADC      |

### 电量监测功能

- **库仑计数 ADC**:支持低至 1mΩ 的电流检测电阻
- **电压 ADC**:用于电池电压和外部/内部温度传感器
- **Impedance Track™ 技术**:
  - 电池放电模拟曲线,精确预测电池续航能力
  - 自动调整电池老化、温度及充放电速率的影响

### 保护功能

BQ27Z746 提供基于硬件的可编程保护:

| 保护类型 |        说明        |
| :------: | :----------------: |
|   OVP    |      过压保护      |
|   UVP    |      欠压保护      |
|   OCD    |    放电过流保护    |
|   OCC    |    充电过流保护    |
|   SCD    |    放电短路保护    |
|    OT    | 基于固件的过热保护 |
|   RCP    |    反向充电保护    |

### 功耗模式

|     模式      | 典型功耗 |
| :-----------: | :------: |
|  SLEEP 模式   |   20µA   |
|   SHIP 模式   |   10µA   |
|   货架模式    |   5µA    |
| SHUTDOWN 模式 |  0.2µA   |

### 封装

超紧凑 15 焊球 NanoFree™ DSBGA 封装

## 二、硬件设计要点

### 1. 典型应用电路

BQ27Z746 支持高侧或低侧电流检测,主要引脚包括:

- **BAT+/BAT-**:电池连接
- **PACK+/PACK-**:系统连接
- **SDA/SCL**:I2C 通信
- **CHG/DSG**:高侧 FET 栅极驱动器输出
- **TS**:温度传感器输入

### 2. 电流检测电阻选择

- 支持低至 **1mΩ** 的检测电阻
- 根据应用电流范围选择合适阻值
- 需要进行电流校准

### 3. 温度传感器

- 支持内部温度传感器
- 支持外部 NTC 热敏电阻

## 三、驱动开发指南

### 1. I2C 通信基础

BQ27Z746 使用标准 I2C 协议通信:

- **设备地址**:0x55 (7位地址)
- **通信速率**:最高 400kHz

### 2. 基本驱动框架 (Linux 内核驱动示例)

```c
#include <linux/i2c.h>
#include <linux/power_supply.h>
#include <linux/module.h>

#define BQ27Z746_I2C_ADDR       0x55

/* 标准命令寄存器 */
#define BQ27Z746_REG_CTRL       0x00    /* Control */
#define BQ27Z746_REG_TEMP       0x06    /* Temperature */
#define BQ27Z746_REG_VOLT       0x08    /* Voltage */
#define BQ27Z746_REG_FLAGS      0x0A    /* Flags */
#define BQ27Z746_REG_NAC        0x0C    /* Nominal Available Capacity */
#define BQ27Z746_REG_FAC        0x0E    /* Full Available Capacity */
#define BQ27Z746_REG_RM         0x10    /* Remaining Capacity */
#define BQ27Z746_REG_FCC        0x12    /* Full Charge Capacity */
#define BQ27Z746_REG_AI         0x14    /* Average Current */
#define BQ27Z746_REG_SOC        0x2C    /* State of Charge */
#define BQ27Z746_REG_SOH        0x2E    /* State of Health */

struct bq27z746_device {
    struct i2c_client *client;
    struct power_supply *battery;
    struct power_supply_desc battery_desc;
};

/* 读取16位寄存器 */
static int bq27z746_read_word(struct i2c_client *client, u8 reg)
{
    int ret;
    ret = i2c_smbus_read_word_data(client, reg);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read reg 0x%02x: %d\n", reg, ret);
        return ret;
    }
    return ret;
}

/* 写入16位寄存器 */
static int bq27z746_write_word(struct i2c_client *client, u8 reg, u16 value)
{
    int ret;
    ret = i2c_smbus_write_word_data(client, reg, value);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to write reg 0x%02x: %d\n", reg, ret);
    }
    return ret;
}

/* 获取电池电压 (mV) */
static int bq27z746_get_voltage(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_VOLT);
    return ret; /* 单位: mV */
}

/* 获取电池温度 (0.1K) */
static int bq27z746_get_temperature(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_TEMP);
    return ret; /* 单位: 0.1K */
}

/* 获取电池电流 (mA) */
static int bq27z746_get_current(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_AI);
    return (s16)ret; /* 有符号值, 单位: mA */
}

/* 获取电池电量百分比 */
static int bq27z746_get_soc(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_SOC);
    return ret; /* 单位: % */
}

/* 获取剩余容量 (mAh) */
static int bq27z746_get_remaining_capacity(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_RM);
    return ret; /* 单位: mAh */
}

/* 获取满充容量 (mAh) */
static int bq27z746_get_full_charge_capacity(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_FCC);
    return ret; /* 单位: mAh */
}

/* 获取电池健康状态 */
static int bq27z746_get_soh(struct bq27z746_device *bq)
{
    int ret = bq27z746_read_word(bq->client, BQ27Z746_REG_SOH);
    return ret & 0xFF; /* 低字节为SOH百分比 */
}

/* Power Supply 属性获取回调 */
static int bq27z746_get_property(struct power_supply *psy,
                                  enum power_supply_property psp,
                                  union power_supply_propval *val)
{
    struct bq27z746_device *bq = power_supply_get_drvdata(psy);
    int ret;

    switch (psp) {
    case POWER_SUPPLY_PROP_VOLTAGE_NOW:
        ret = bq27z746_get_voltage(bq);
        if (ret < 0) return ret;
        val->intval = ret * 1000; /* 转换为 uV */
        break;
    case POWER_SUPPLY_PROP_CURRENT_NOW:
        ret = bq27z746_get_current(bq);
        if (ret < 0) return ret;
        val->intval = ret * 1000; /* 转换为 uA */
        break;
    case POWER_SUPPLY_PROP_CAPACITY:
        ret = bq27z746_get_soc(bq);
        if (ret < 0) return ret;
        val->intval = ret;
        break;
    case POWER_SUPPLY_PROP_TEMP:
        ret = bq27z746_get_temperature(bq);
        if (ret < 0) return ret;
        val->intval = ret - 2730; /* 转换为 0.1°C */
        break;
    case POWER_SUPPLY_PROP_CHARGE_FULL:
        ret = bq27z746_get_full_charge_capacity(bq);
        if (ret < 0) return ret;
        val->intval = ret * 1000; /* 转换为 uAh */
        break;
    case POWER_SUPPLY_PROP_CHARGE_NOW:
        ret = bq27z746_get_remaining_capacity(bq);
        if (ret < 0) return ret;
        val->intval = ret * 1000; /* 转换为 uAh */
        break;
    case POWER_SUPPLY_PROP_HEALTH:
        ret = bq27z746_get_soh(bq);
        if (ret < 0) return ret;
        if (ret >= 80)
            val->intval = POWER_SUPPLY_HEALTH_GOOD;
        else
            val->intval = POWER_SUPPLY_HEALTH_DEAD;
        break;
    default:
        return -EINVAL;
    }
    return 0;
}

static enum power_supply_property bq27z746_props[] = {
    POWER_SUPPLY_PROP_VOLTAGE_NOW,
    POWER_SUPPLY_PROP_CURRENT_NOW,
    POWER_SUPPLY_PROP_CAPACITY,
    POWER_SUPPLY_PROP_TEMP,
    POWER_SUPPLY_PROP_CHARGE_FULL,
    POWER_SUPPLY_PROP_CHARGE_NOW,
    POWER_SUPPLY_PROP_HEALTH,
};

static int bq27z746_probe(struct i2c_client *client,
                          const struct i2c_device_id *id)
{
    struct bq27z746_device *bq;
    struct power_supply_config psy_cfg = {};

    bq = devm_kzalloc(&client->dev, sizeof(*bq), GFP_KERNEL);
    if (!bq)
        return -ENOMEM;

    bq->client = client;
    i2c_set_clientdata(client, bq);

    bq->battery_desc.name = "bq27z746-battery";
    bq->battery_desc.type = POWER_SUPPLY_TYPE_BATTERY;
    bq->battery_desc.properties = bq27z746_props;
    bq->battery_desc.num_properties = ARRAY_SIZE(bq27z746_props);
    bq->battery_desc.get_property = bq27z746_get_property;

    psy_cfg.drv_data = bq;
    psy_cfg.of_node = client->dev.of_node;

    bq->battery = devm_power_supply_register(&client->dev,
                                              &bq->battery_desc, &psy_cfg);
    if (IS_ERR(bq->battery)) {
        dev_err(&client->dev, "Failed to register power supply\n");
        return PTR_ERR(bq->battery);
    }

    dev_info(&client->dev, "BQ27Z746 battery gauge probed\n");
    return 0;
}

static const struct i2c_device_id bq27z746_id[] = {
    { "bq27z746", 0 },
    { }
};
MODULE_DEVICE_TABLE(i2c, bq27z746_id);

static const struct of_device_id bq27z746_of_match[] = {
    { .compatible = "ti,bq27z746" },
    { }
};
MODULE_DEVICE_TABLE(of, bq27z746_of_match);

static struct i2c_driver bq27z746_driver = {
    .driver = {
        .name = "bq27z746",
        .of_match_table = bq27z746_of_match,
    },
    .probe = bq27z746_probe,
    .id_table = bq27z746_id,
};
module_i2c_driver(bq27z746_driver);

MODULE_DESCRIPTION("BQ27Z746 Battery Gauge Driver");
MODULE_LICENSE("GPL v2");
```

## 三、驱动代码实现

### 1. MAC 命令读取函数

```c
#define BQ27Z746_REG_MAC           0x3E    /* AltManufacturerAccess */
#define BQ27Z746_REG_MAC_DATA      0x40    /* MAC Data 起始地址 */
#define BQ27Z746_REG_MAC_DATA_SUM  0x60    /* MAC Data 校验和 */
#define BQ27Z746_REG_MAC_DATA_LEN  0x61    /* MAC Data 长度 */

/* MAC 命令定义 */
#define BQ27Z746_MAC_SAFETY_ALERT       0x0050
#define BQ27Z746_MAC_SAFETY_STATUS      0x0051
#define BQ27Z746_MAC_PF_ALERT           0x0052
#define BQ27Z746_MAC_PF_STATUS          0x0053
#define BQ27Z746_MAC_OPERATION_STATUS   0x0054
#define BQ27Z746_MAC_CHARGING_STATUS    0x0055
#define BQ27Z746_MAC_GAUGING_STATUS     0x0056
#define BQ27Z746_MAC_MFG_STATUS         0x0057
#define BQ27Z746_MAC_PROTECTOR_STATUS   0x0058
#define BQ27Z746_MAC_BATT_SENSE_STATUS  0x005A

/**
 * bq27z746_mac_read - 通过 MAC 命令读取数据
 * @client: I2C 客户端
 * @cmd: MAC 命令
 * @data: 数据缓冲区
 * @len: 期望读取的数据长度
 * 
 * 返回: 0 成功, 负值 失败
 */
static int bq27z746_mac_read(struct i2c_client *client, u16 cmd, 
                              u8 *data, int len)
{
    int ret;
    u8 buf[36];
    u8 checksum, read_len;
    int i;

    /* 步骤1: 写入 MAC 命令到 0x3E (小端序) */
    buf[0] = cmd & 0xFF;
    buf[1] = (cmd >> 8) & 0xFF;
    ret = i2c_smbus_write_i2c_block_data(client, BQ27Z746_REG_MAC, 2, buf);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to write MAC cmd 0x%04x\n", cmd);
        return ret;
    }

    /* 步骤2: 等待数据准备就绪 */
    usleep_range(1000, 2000);

    /* 步骤3: 读取 MAC Data 长度 */
    ret = i2c_smbus_read_byte_data(client, BQ27Z746_REG_MAC_DATA_LEN);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read MAC data length\n");
        return ret;
    }
    read_len = ret;

    if (read_len < 4 || read_len > 36) {
        dev_err(&client->dev, "Invalid MAC data length: %d\n", read_len);
        return -EINVAL;
    }

    /* 步骤4: 读取完整的 MAC Data (包括命令回显) */
    ret = i2c_smbus_read_i2c_block_data(client, BQ27Z746_REG_MAC, 
                                         read_len, buf);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read MAC data\n");
        return ret;
    }

    /* 步骤5: 验证命令回显 */
    if (buf[0] != (cmd & 0xFF) || buf[1] != ((cmd >> 8) & 0xFF)) {
        dev_err(&client->dev, "MAC cmd echo mismatch\n");
        return -EIO;
    }

    /* 步骤6: 读取并验证校验和 */
    ret = i2c_smbus_read_byte_data(client, BQ27Z746_REG_MAC_DATA_SUM);
    if (ret < 0) {
        dev_err(&client->dev, "Failed to read MAC checksum\n");
        return ret;
    }
    
    checksum = 0;
    for (i = 0; i < read_len; i++)
        checksum += buf[i];
    checksum = 0xFF - checksum;
    
    if (checksum != (u8)ret) {
        dev_err(&client->dev, "MAC checksum error: calc=0x%02x, read=0x%02x\n",
                checksum, ret);
        return -EIO;
    }

    /* 步骤7: 复制有效数据 (跳过2字节命令回显) */
    if (len > read_len - 2)
        len = read_len - 2;
    memcpy(data, &buf[2], len);

    return 0;
}

/**
 * bq27z746_mac_read_u32 - 读取32位 MAC 数据
 */
static int bq27z746_mac_read_u32(struct i2c_client *client, u16 cmd, u32 *val)
{
    u8 data[4];
    int ret;

    ret = bq27z746_mac_read(client, cmd, data, 4);
    if (ret < 0)
        return ret;

    /* 小端序转换 */
    *val = data[0] | (data[1] << 8) | (data[2] << 16) | (data[3] << 24);
    return 0;
}
```

```c
struct bq27z746_protection_config {
    /* 固件保护 */
    u16 cuv_threshold;      /* 欠压阈值 (mV) */

    u16 cov_delay;          /* 过压延迟 (ms) */
    u16 cov_recovery;       /* 过压恢复 (mV) */

    u16 otd_delay;          /* 放电过温延迟 (s) */
    s16 otd_recovery;       /* 放电过温恢复 (0.1°C) */

    u16 hcuv_threshold;     /* 硬件欠压阈值 */
    u16 hcov_threshold;     /* 硬件过压阈值 */
    u16 hocc_threshold;     /* 硬件充电过流阈值 */
    u16 hocd_threshold;     /* 硬件放电过流阈值 */
    u16 hscd_threshold;     /* 硬件短路阈值 */

 * bq27z746_read_protection_config - 读取所有保护配置
 */
static int bq27z746_read_protection_config(struct bq27z746_device *bq,
                                            struct bq27z746_protection_config *cfg)
{
    int ret;

    /* 读取固件保护配置 */
    ret = bq27z746_read_df_word(bq->client, DF_CUV_THRESHOLD, &cfg->cuv_threshold);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_CUV_DELAY, &cfg->cuv_delay);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_CUV_RECOVERY, &cfg->cuv_recovery);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_COV_THRESHOLD, &cfg->cov_threshold);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_COV_DELAY, &cfg->cov_delay);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_COV_RECOVERY, &cfg->cov_recovery);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_OCC_THRESHOLD, &cfg->occ_threshold);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_OCD_THRESHOLD, &cfg->ocd_threshold);
    if (ret) return ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_SCD_THRESHOLD, &cfg->scd_threshold);
    if (ret) return ret;

    /* 读取硬件保护配置 */
    ret = bq27z746_read_df_word(bq->client, DF_HCUV_THRESHOLD, &cfg->hcuv_threshold);
    if (ret) return ret;
```

```c
 * bq27z746_get_all_fault_status - 获取所有故障状态
 */
static int bq27z746_get_all_fault_status(struct bq27z746_device *bq,
                                          struct bq27z746_fault_status *fault)
{
    int ret;

    /* 读取 BatteryStatus */
    ret = bq27z746_read_word(bq->client, BQ27Z746_REG_FLAGS);
    if (ret < 0) return ret;
    fault->battery_status = ret;

    /* 读取 SafetyAlert */
    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_SAFETY_ALERT, 
                                 &fault->safety_alert);
    if (ret) return ret;

    /* 读取 SafetyStatus */
    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_SAFETY_STATUS, 
                                 &fault->safety_status);
    if (ret) return ret;

    /* 读取 PFAlert */
    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_PF_ALERT, 
                                 &fault->pf_alert);
    if (ret) return ret;

    /* 读取 PFStatus */
    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_PF_STATUS, 
                                 &fault->pf_status);
    if (ret) return ret;

    /* 读取 OperationStatus */
    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_OPERATION_STATUS, 
                                 &fault->operation_status);
    if (ret) return ret;

    /* 读取 ProtectorStatus */
    ret = bq27z746_get_protector_status(bq, &fault->protector_status);
    if (ret) return ret;

    /* 读取 BattSenseStatus */
    ret = bq27z746_get_batt_sense_status(bq, &fault->batt_sense_status);
    if (ret) return ret;

    return 0;
}

/**
 * bq27z746_print_fault_status - 打印故障状态 (调试用)
 */
static void bq27z746_print_fault_status(struct bq27z746_device *bq,
                                         struct bq27z746_fault_status *fault)
{
    dev_info(&bq->client->dev, "=== BQ27Z746 Fault Status ===\n");
    dev_info(&bq->client->dev, "BatteryStatus:    0x%04X\n", fault->battery_status);
    dev_info(&bq->client->dev, "SafetyAlert:      0x%08X\n", fault->safety_alert);
    dev_info(&bq->client->dev, "SafetyStatus:     0x%08X\n", fault->safety_status);
    dev_info(&bq->client->dev, "PFAlert:          0x%08X\n", fault->pf_alert);
    dev_info(&bq->client->dev, "PFStatus:         0x%08X\n", fault->pf_status);
    dev_info(&bq->client->dev, "OperationStatus:  0x%08X\n", fault->operation_status);
    dev_info(&bq->client->dev, "ProtectorStatus:  0x%04X\n", fault->protector_status);
    dev_info(&bq->client->dev, "BattSenseStatus:  0x%02X\n", fault->batt_sense_status);

    /* 解析具体故障 */
    if (fault->safety_status & SAFETY_CUV)
        dev_warn(&bq->client->dev, "  -> Cell Undervoltage!\n");
    if (fault->safety_status & SAFETY_COV)
        dev_warn(&bq->client->dev, "  -> Cell Overvoltage!\n");
    if (fault->safety_status & SAFETY_OCC)
        dev_warn(&bq->client->dev, "  -> Overcurrent in Charge!\n");
    if (fault->safety_status & SAFETY_OCD)
        dev_warn(&bq->client->dev, "  -> Overcurrent in Discharge!\n");
    if (fault->safety_status & SAFETY_OTC)
        dev_warn(&bq->client->dev, "  -> Overtemperature in Charge!\n");
    if (fault->safety_status & SAFETY_OTD)
        dev_warn(&bq->client->dev, "  -> Overtemperature in Discharge!\n");
    if (fault->safety_status & SAFETY_ASCD)
        dev_warn(&bq->client->dev, "  -> Short Circuit in Discharge!\n");

    /* 硬件保护状态 */
    if (fault->protector_status & PROT_HCUV)
        dev_warn(&bq->client->dev, "  -> HW Cell Undervoltage!\n");
    if (fault->protector_status & PROT_HCOV)
        dev_warn(&bq->client->dev, "  -> HW Cell Overvoltage!\n");
    if (fault->protector_status & PROT_HSCD)

}
```

```c
/* MAC 控制命令 */
#define BQ27Z746_MAC_RESET              0x0041

/**
 * bq27z746_mac_command - 发送 MAC 控制命令

static int bq27z746_mac_command(struct i2c_client *client, u16 cmd)
{
    u8 buf[2];

    return i2c_smbus_write_i2c_block_data(client, BQ27Z746_REG_MAC, 2, buf);
}
```

# BQ27Z746 功耗模式详解及设定

## 一、功耗模式概览

BQ27Z746 支持 5 种功耗模式，从高到低依次为：

|     模式     |  典型功耗  | FET 状态 | I2C 通信 | 电量计算 | 主要用途  |
| :----------: | :--------: | :------: | :------: | :------: | :-------: |
|  **NORMAL**  | ~100-200µA |   开启   |   可用   |   活跃   | 正常工作  |
|  **SLEEP**   |   ~20µA    |   开启   |   可用   | 降低频率 | 待机/空闲 |
|   **SHIP**   |   ~10µA    |   关闭   |   可用   |   暂停   | 运输/存储 |
|  **SHELF**   |    ~5µA    |   关闭   |   可用   |   暂停   | 长期存储  |
| **SHUTDOWN** |   ~0.2µA   |   关闭   |  不可用  |   停止   | 深度休眠  |

```text
功耗递减方向:
NORMAL → SLEEP → SHIP → SHELF → SHUTDOWN
  ↑                                    ↓
  └────────────────────────────────────┘
              唤醒方向
```

------

## 二、各模式详细说明

### 1. NORMAL 模式 (正常模式)

**特点：**

- 电量计完全活跃，持续监测电池状态
- 所有测量功能以最高频率运行
- FET 正常控制充放电
- I2C 通信完全可用

**进入条件：**

- 上电默认进入
- 从其他模式唤醒后进入
- 检测到电流 > SleepCurrent

**退出条件：**

- 电流 < SleepCurrent 且持续 Sleep Delay 时间后自动进入 SLEEP

------

### 2. SLEEP 模式 (睡眠模式)

**特点：**

- 电量计以降低的频率运行
- 周期性唤醒进行测量
- FET 保持开启状态
- I2C 通信可用（会唤醒芯片）
- 保护功能仍然有效

**进入条件：**

- 电流 < SleepCurrent 且持续 Sleep Delay 时间
- 自动进入（无需命令）

**退出条件：**

- 检测到电流 > SleepCurrent（任意方向）
- 收到 I2C 通信
- 发生保护事件

**相关配置参数：**

|     参数     | Data Flash 地址 |          说明           |
| :----------: | :-------------: | :---------------------: |
| SleepCurrent |     0x4600      | 进入睡眠的电流阈值 (mA) |
| Sleep Delay  |     0x4602      | 进入睡眠的延迟时间 (s)  |

------

### 3. SHIP 模式 (运输模式)

**特点：**

- 用于产品运输和短期存储
- CHG 和 DSG FET 关闭，断开电池与系统连接
- 电量计暂停大部分功能
- I2C 通信仍可用
- 硬件保护仍然有效

**进入条件：**

- 发送 SHIP_MODE MAC 命令 (0x0010)
- 或通过 ENAB 引脚触发（如果配置）

**退出条件：**

- 检测到充电电压（连接充电器）
- 检测到电流 > IWAKE（如果启用 IWAKE 功能）
- 发送 EXIT_SHIP MAC 命令
- ENAB 引脚状态变化（如果配置）

**相关配置参数：**

|       参数        | Data Flash 地址 |     说明     |
| :---------------: | :-------------: | :----------: |
|  IWAKE Threshold  |     0x4610      | 唤醒电流阈值 |
| Ship FET Off Time |     0x4612      | FET 关闭延迟 |
|   Power Config    |     0x4620      |  电源配置位  |

------

### 4. SHELF 模式 (货架模式)

**特点：**

- 用于长期存储
- 比 SHIP 模式功耗更低
- CHG 和 DSG FET 关闭
- 电量计几乎完全暂停
- I2C 通信可用但响应较慢
- 只有充电电压检测可以唤醒

**进入条件：**

- 发送 SHELF_MODE MAC 命令 (0x0011)

**退出条件：**

- 检测到充电电压（连接充电器）
- 发送 EXIT_SHELF MAC 命令
- **注意：电流唤醒 (IWAKE) 在 SHELF 模式下不工作**

------

### 5. SHUTDOWN 模式 (关机模式)

**特点：**

- 最低功耗模式
- 几乎所有功能停止
- FET 关闭
- I2C 通信不可用
- 只能通过硬件方式唤醒

**进入条件：**

- 发送 SHUTDOWN MAC 命令 (0x0012)
- 或电池电压低于 Shutdown Voltage

**退出条件：**

- 检测到充电电压
- ENAB 引脚状态变化
- **I2C 命令无法唤醒**

```c
 * @cmd: MAC 命令
 * 
 * 返回: 0 成功, 负值 失败
 */
static int bq27z746_mac_command(struct i2c_client *client, u16 cmd)
{
    u8 buf[2];
    int ret;
    
    buf[0] = cmd & 0xFF;
    buf[1] = (cmd >> 8) & 0xFF;
    
    ret = i2c_smbus_write_i2c_block_data(client, BQ27Z746_REG_MAC, 2, buf);

        dev_err(&client->dev, "Failed to send MAC cmd 0x%04x: %d\n", cmd, ret);
        return ret;
    }
    
    /* 等待命令执行 */
    usleep_range(1000, 2000);
    
    return 0;
}
```

### 4. 写入 Data Flash 配置

```c
/**
 * bq27z746_write_df_word - 写入 Data Flash 16位值
 * 
 * 注意: 写入 Data Flash 需要先进入 CONFIG UPDATE 模式
 */
static int bq27z746_write_df_word(struct i2c_client *client, u16 addr, u16 val)
{
    u8 buf[4];
    u8 checksum;
    int ret;

    /* 写入地址和数据到 MAC */
    buf[0] = addr & 0xFF;
    buf[1] = (addr >> 8) & 0xFF;
    buf[2] = val & 0xFF;
    buf[3] = (val >> 8) & 0xFF;

    ret = i2c_smbus_write_i2c_block_data(client, BQ27Z746_REG_MAC, 4, buf);
    if (ret < 0)
        return ret;

    /* 计算并写入校验和 */
    checksum = 0xFF - (buf[0] + buf[1] + buf[2] + buf[3]);
    ret = i2c_smbus_write_byte_data(client, BQ27Z746_REG_MAC_DATA_SUM, checksum);
    if (ret < 0)
        return ret;

    /* 写入长度 */
    ret = i2c_smbus_write_byte_data(client, BQ27Z746_REG_MAC_DATA_LEN, 4 + 2);
    if (ret < 0)
        return ret;

    /* 等待写入完成 */
    msleep(10);

    return 0;
}

/**
 * bq27z746_enter_config_update - 进入配置更新模式
 */
static int bq27z746_enter_config_update(struct bq27z746_device *bq)
{
    int ret;
    u32 op_status;
    int timeout = 100;

    /* 发送 SET_CFGUPDATE 命令 */
    ret = bq27z746_mac_command(bq->client, 0x0090);
    if (ret < 0)
        return ret;

    /* 等待进入 CONFIG UPDATE 模式 */
    while (timeout--) {
        msleep(10);
        ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_OPERATION_STATUS, 
                                     &op_status);
        if (ret < 0)
            return ret;

        if (op_status & OPSTATUS_CFGUPDATE)
            return 0;
    }

    dev_err(&bq->client->dev, "Timeout entering CONFIG UPDATE mode\n");
    return -ETIMEDOUT;
}

/**
 * bq27z746_exit_config_update - 退出配置更新模式
 */
static int bq27z746_exit_config_update(struct bq27z746_device *bq)
{
    int ret;
    u32 op_status;
    int timeout = 100;

    /* 发送 EXIT_CFGUPDATE 命令 */
    ret = bq27z746_mac_command(bq->client, 0x0092);
    if (ret < 0)
        return ret;

    /* 等待退出 CONFIG UPDATE 模式 */
    while (timeout--) {
        msleep(10);
        ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_OPERATION_STATUS, 
                                     &op_status);
        if (ret < 0)
            return ret;

        if (!(op_status & OPSTATUS_CFGUPDATE))
            return 0;
    }

    dev_err(&bq->client->dev, "Timeout exiting CONFIG UPDATE mode\n");
    return -ETIMEDOUT;
}

/**
 * bq27z746_set_sleep_current - 设置 Sleep 电流阈值
 */
static int bq27z746_set_sleep_current(struct bq27z746_device *bq, u16 current_ma)
{
    int ret;

    ret = bq27z746_enter_config_update(bq);
    if (ret)
        return ret;

    ret = bq27z746_write_df_word(bq->client, DF_SLEEP_CURRENT, current_ma);
    if (ret) {
        bq27z746_exit_config_update(bq);
        return ret;
    }

    return bq27z746_exit_config_update(bq);
}
```

## 九、注意事项

### 1. SHIP 模式

- 进入后 FET 关闭，系统将断电
- 设置 `SHIPM_ON_RST` 位可防止意外复位导致退出
- IWAKE 功能需要在 Power Config 中启用

### 2. SHELF 模式

- 比 SHIP 模式功耗更低
- **IWAKE 功能不工作**，只能通过充电电压唤醒
- 适合长期存储（数月）

### 3. SHUTDOWN 模式

- **I2C 通信不可用**
- 只能通过硬件方式（充电电压或 ENAB 引脚）唤醒
- 进入前确保有唤醒手段

### 4. SLEEP 模式

- 自动进入和退出
- 配置合适的 SleepCurrent 阈值很重要
- 阈值过低可能导致无法进入 SLEEP
- 阈值过高可能导致频繁进入/退出

## 二、三种安全访问模式

### 1. 模式概览

|      模式       | 安全级别 |      主要用途      |     权限范围     |
| :-------------: | :------: | :----------------: | :--------------: |
|   **SEALED**    |   最高   | 产品出厂后正常使用 |   只读基本数据   |
|  **UNSEALED**   |   中等   | 开发调试、配置修改 |  读写大部分配置  |
| **FULL ACCESS** |   最低   | 工厂生产、固件更新 | 完全访问所有功能 |

### 2. 模式状态检测

通过 OperationStatus (MAC 0x0054) 寄存器的 SEC0 和 SEC1 位判断当前模式：

| SEC1 | SEC0 |  安全模式   |
| :--: | :--: | :---------: |
|  0   |  0   | FULL ACCESS |
|  0   |  1   |  UNSEALED   |
|  1   |  0   |   SEALED    |
|  1   |  1   |   SEALED    |

```c
/* 安全模式位定义 */
#define OPSTATUS_SEC0           BIT(11)  /* 安全模式位0 */
#define OPSTATUS_SEC1           BIT(12)  /* 安全模式位1 */

/* 安全模式枚举 */
enum bq27z746_security_mode {
    BQ27Z746_SEC_FULL_ACCESS = 0,   /* SEC1=0, SEC0=0 */
    BQ27Z746_SEC_UNSEALED    = 1,   /* SEC1=0, SEC0=1 */
    BQ27Z746_SEC_SEALED      = 2,   /* SEC1=1, SEC0=0 或 SEC1=1, SEC0=1 */
};

/**
 * bq27z746_get_security_mode - 获取当前安全模式
 */
static int bq27z746_get_security_mode(struct bq27z746_device *bq,
                                       enum bq27z746_security_mode *mode)
{
    u32 op_status;
    int ret;

    ret = bq27z746_mac_read_u32(bq->client, BQ27Z746_MAC_OPERATION_STATUS, 
                                 &op_status);
    if (ret < 0)
        return ret;

    if (op_status & OPSTATUS_SEC1) {
        *mode = BQ27Z746_SEC_SEALED;
    } else if (op_status & OPSTATUS_SEC0) {
        *mode = BQ27Z746_SEC_UNSEALED;
    } else {
        *mode = BQ27Z746_SEC_FULL_ACCESS;
    }

    return 0;
}

/**
 * bq27z746_get_security_mode_string - 获取模式名称
 */
static const char *bq27z746_get_security_mode_string(enum bq27z746_security_mode mode)
{
    static const char *mode_names[] = {
        [BQ27Z746_SEC_FULL_ACCESS] = "FULL ACCESS",
        [BQ27Z746_SEC_UNSEALED]    = "UNSEALED",
        [BQ27Z746_SEC_SEALED]      = "SEALED",
    };
    return mode_names[mode];
}
```

## 三、各模式权限详解

### 1. SEALED 模式 (密封模式)

**特点：**

- 产品出厂后的默认运行模式
- 保护配置数据不被篡改
- 只允许读取基本电池信息

**可访问的功能：**

|    功能类别    | 可读 | 可写 |            说明             |
| :------------: | :--: | :--: | :-------------------------: |
| 电压/电流/温度 |  ✓   |  ✗   |        基本测量数据         |
|    SOC/SOH     |  ✓   |  ✗   |       电量和健康状态        |
|    容量信息    |  ✓   |  ✗   |         RM, FCC 等          |
|    状态标志    |  ✓   |  ✗   | BatteryStatus, SafetyStatus |
|    设备信息    |  ✓   |  ✗   |     制造商信息、序列号      |
|   Data Flash   |  ✗   |  ✗   |      配置参数不可访问       |
|    保护配置    |  ✗   |  ✗   |      保护阈值不可修改       |
|    FET 控制    |  ✗   |  ✗   |      不能手动控制 FET       |
|  SHA-256 认证  |  ✓   |  -   |       可进行电池认证        |

### 2. UNSEALED 模式 (解封模式)

**特点：**

- 开发和调试阶段使用

## 五、模式切换操作

### 1. SEALED → UNSEALED (解封)

```c
/**
 * bq27z746_unseal - 从 SEALED 模式解封到 UNSEALED 模式
 * @bq: 设备结构体
 * @key_first: Unseal Key 第一部分
 * @key_second: Unseal Key 第二部分
 * 
 * 解封过程需要按顺序发送两个16位密钥
 */
static int bq27z746_unseal(struct bq27z746_device *bq, 
                            u16 key_first, u16 key_second)
{
    int ret;
    enum bq27z746_security_mode mode;

    /* 检查当前模式 */
    ret = bq27z746_get_security_mode(bq, &mode);
    if (ret)
        return ret;

    if (mode != BQ27Z746_SEC_SEALED) {
        dev_info(&bq->client->dev, "Already in %s mode\n",
                 bq27z746_get_security_mode_string(mode));
        return 0;
    }

    /* 发送第一部分密钥 */
    ret = bq27z746_mac_command(bq->client, key_first);
    if (ret) {
        dev_err(&bq->client->dev, "Failed to send unseal key first\n");
        return ret;
    }

    /* 短暂延迟 */
    usleep_range(1000, 2000);

    /* 发送第二部分密钥 */
    ret = bq27z746_mac_command(bq->client, key_second);
    if (ret) {
        dev_err(&bq->client->dev, "Failed to send unseal key second\n");
        return ret;
    }

    /* 等待模式切换 */
    msleep(10);

    /* 验证模式切换 */
    ret = bq27z746_get_security_mode(bq, &mode);
    if (ret)
        return ret;

    if (mode == BQ27Z746_SEC_UNSEALED) {
```

## 六、SHA-256 电池认证

### 1. 认证原理

BQ27Z746 集成 SHA-256 认证响应器，用于系统验证电池包的真实性：

```text
┌─────────────────────────────────────────────────────────────────┐
│                    SHA-256 挑战-响应认证                        │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│   主机 (Host)                      电池 (BQ27Z746)              │
│       │                                  │                      │
│       │  1. 发送随机挑战 (Challenge)     │                      │
│       │ ─────────────────────────────────>                      │
│       │     (20 bytes 随机数)            │                      │
│       │                                  │                      │
│       │                    2. 计算 SHA-256                      │
│       │                    Hash = SHA256(Key + Challenge)       │
│       │                                  │                      │
│       │  3. 返回响应 (Response)          │                      │
│       │ <─────────────────────────────────                      │
│       │     (32 bytes Hash)              │                      │
│       │                                  │                      │
│   4. 验证响应                            │                      │
│   Expected = SHA256(Key + Challenge)     │                      │
│   if (Response == Expected)              │                      │
│       电池认证通过                       │                      │
│   else                                   │                      │
│       电池认证失败 (可能是假冒电池)      │                      │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 2. 认证相关 MAC 命令

```c
/* SHA-256 认证相关命令 */
#define BQ27Z746_MAC_AUTHEN_FIRST       0x0044  /* 认证第一步 */
#define BQ27Z746_MAC_AUTHEN_SECOND      0x0045  /* 认证第二步 */

/* 认证数据长度 */
#define SHA256_CHALLENGE_LEN    20  /* 挑战数据长度 (bytes) */
#define SHA256_RESPONSE_LEN     32  /* 响应数据长度 (bytes) */
#define SHA256_KEY_LEN          16  /* 认证密钥长度 (bytes) */
```

### 3. 认证流程实现

```c
/**
 * bq27z746_authenticate - 执行 SHA-256 电池认证
 * @bq: 设备结构体
 * @challenge: 20字节随机挑战数据
 * @response: 32字节响应数据输出缓冲区
 * 
 * 返回: 0 成功, 负值 失败
 */
static int bq27z746_authenticate(struct bq27z746_device *bq,
                                  const u8 *challenge, u8 *response)
{
    u8 buf[36];
    u8 checksum;
    int ret, i;

    /* 步骤1: 发送挑战数据 */
    /* 写入 MAC 命令 + 挑战数据 */
    buf[0] = BQ27Z746_MAC_AUTHEN_FIRST & 0xFF;
    buf[1] = (BQ27Z746_MAC_AUTHEN_FIRST >> 8) & 0xFF;
    memcpy(&buf[2], challenge, SHA256_CHALLENGE_LEN);

    /* 计算校验和 */
    checksum = 0;
    for (i = 0; i < 2 + SHA256_CHALLENGE_LEN; i++)
        checksum += buf[i];
    checksum = 0xFF - checksum;

    /* 写入数据 */
    ret = i2c_smbus_write_i2c_block_data(bq->client, BQ27Z746_REG_MAC,
                                          2 + SHA256_CHALLENGE_LEN, buf);
    if (ret < 0)
        return ret;

    /* 写入校验和和长度 */
    ret = i2c_smbus_write_byte_data(bq->client, BQ27Z746_REG_MAC_DATA_SUM, 
                                     checksum);
    if (ret < 0)
        return ret;

    ret = i2c_smbus_write_byte_data(bq->client, BQ27Z746_REG_MAC_DATA_LEN,
                                     2 + SHA256_CHALLENGE_LEN + 2);
    if (ret < 0)
        return ret;

    /* 等待计算完成 */
    msleep(50);

    /* 步骤2: 读取响应数据 */
    ret = bq27z746_mac_read(bq->client, BQ27Z746_MAC_AUTHEN_SECOND,
                             response, SHA256_RESPONSE_LEN);
    if (ret < 0)
        return ret;

    return 0;
}

/**
 * bq27z746_verify_battery - 验证电池真实性
 * @bq: 设备结构体
 * @auth_key: 16字节认证密钥 (与电池中存储的密钥相同)
 * 
 * 返回: 0 认证通过, -EACCES 认证失败, 其他负值 通信错误
 */
static int bq27z746_verify_battery(struct bq27z746_device *bq,
                                    const u8 *auth_key)
{
    u8 challenge[SHA256_CHALLENGE_LEN];
    u8 response[SHA256_RESPONSE_LEN];
    u8 expected[SHA256_RESPONSE_LEN];
    u8 hash_input[SHA256_KEY_LEN + SHA256_CHALLENGE_LEN];
    struct crypto_shash *tfm;
    struct shash_desc *desc;
    int ret;

    /* 生成随机挑战 */
    get_random_bytes(challenge, SHA256_CHALLENGE_LEN);

    /* 发送挑战并获取响应 */
    ret = bq27z746_authenticate(bq, challenge, response);
    if (ret < 0)
        return ret;

    /* 在主机端计算期望的响应 */
    /* Hash = SHA256(Key || Challenge) */
    memcpy(hash_input, auth_key, SHA256_KEY_LEN);
    memcpy(hash_input + SHA256_KEY_LEN, challenge, SHA256_CHALLENGE_LEN);

    tfm = crypto_alloc_shash("sha256", 0, 0);
    if (IS_ERR(tfm))
        return PTR_ERR(tfm);

    desc = kzalloc(sizeof(*desc) + crypto_shash_descsize(tfm), GFP_KERNEL);
    if (!desc) {
        crypto_free_shash(tfm);
        return -ENOMEM;
    }

    desc->tfm = tfm;
    ret = crypto_shash_digest(desc, hash_input, 
                               SHA256_KEY_LEN + SHA256_CHALLENGE_LEN,
                               expected);
    
    kfree(desc);
    crypto_free_shash(tfm);

    if (ret < 0)
        return ret;

    /* 比较响应 */
    if (memcmp(response, expected, SHA256_RESPONSE_LEN) == 0) {
        dev_info(&bq->client->dev, "Battery authentication PASSED\n");
        return 0;
    } else {
        dev_warn(&bq->client->dev, "Battery authentication FAILED!\n");
        return -EACCES;
    }
}
```

```c
        return ret;

    /* 5. 密封设备 */

}
```

### 2. MAC 命令详细流程

```c

 * MAC 命令访问的完整流程
 */

/* 步骤1: 写入 MAC 命令 (16位) 到地址 0x3E */
i2c_write(0x3E, [cmd_low, cmd_high]);

/* 步骤2: 等待处理完成 */
delay(1-50ms);  // 根据命令类型不同

/* 步骤3: 读取数据长度 (从 0x61) */
len = i2c_read(0x61);

/* 步骤4: 读取数据 (从 0x40 开始) */
data = i2c_read_block(0x40, len);

/* 步骤5: 验证校验和 (从 0x60) */
checksum = i2c_read(0x60);
verify(checksum, data);

/* 为什么这么复杂？因为: */
/* 1. 某些命令需要内部计算 (如 SHA-256) */
/* 2. 某些命令需要访问 Flash (较慢) */
/* 3. 数据长度可变，需要先知道长度 */
/* 4. 需要校验和保证数据完整性 */
```

## 四、为什么不全用 MAC 命令？

### 1. 性能考虑

```c
/* 读取电压的两种方式对比 */

/* 方式1: 直接寄存器 (推荐用于频繁访问的数据) */
int read_voltage_direct(void) {
    return i2c_read_word(0x08);  // 1次 I2C 事务, ~100µs
}

/* 方式2: MAC 命令 (不推荐用于频繁访问) */
int read_voltage_mac(void) {
    i2c_write_word(0x3E, 0x0008);  // 写命令
    delay(1000);                    // 等待
    return i2c_read_word(0x40);    // 读数据
    // 3次 I2C 事务 + 延迟, ~2ms
}

/* 如果每秒读取10次电压: */
/* 直接寄存器: 10 × 100µs = 1ms */
/* MAC 命令:   10 × 2ms = 20ms */
/* 性能差 20 倍！ */
```

### 1. 安全性

```c

 * 1. 阻值范围: 1mΩ ~ 10mΩ (推荐 1mΩ ~ 5mΩ)

 *    最大电流 = 32.768mV / Rsense

 */

/* 推荐配置 */

    int max_current_ma;   /* 最大可测电流 (mA) */

    { 1,  32768, "大电流应用 (>10A)" },

    { 10, 3276,  "超低电流应用 (<3A)" },
```

### 1. I2C 通信注意事项

```c
/*
 * I2C 通信关键注意事项
 */

/* 1. 通信速率 */
#define BQ27Z746_I2C_MAX_FREQ   400000  /* 最大 400kHz */
/* 建议: 开发阶段使用 100kHz，稳定后再提高到 400kHz */

/* 2. 从 SLEEP 模式唤醒 */
/*
 * 当设备处于 SLEEP 模式时，第一次 I2C 通信可能失败!
 * 
 * 原因: I2C 通信会唤醒设备，但唤醒需要时间
 * 解决: 发送一次"唤醒"通信，等待后再进行实际通信
 */
static int bq27z746_wake_and_read(struct bq27z746_device *bq, u8 reg)
{
    int ret;
    
    /* 第一次读取用于唤醒 (可能失败，忽略错误) */
    i2c_smbus_read_word_data(bq->client, reg);
    
    /* 等待设备唤醒 */
    usleep_range(2000, 3000);
    
    /* 实际读取 */
    ret = i2c_smbus_read_word_data(bq->client, reg);
    
    return ret;
}

/* 3. MAC 命令超时处理 */
/*
 * 某些 MAC 命令需要较长处理时间:
 * - 普通命令: 1-2ms
 * - Flash 操作: 10-50ms
 * - SHA-256 计算: 50-100ms
 * - 配置更新: 100-500ms
 */
static int bq27z746_mac_command_with_timeout(struct bq27z746_device *bq,
                                              u16 cmd, int timeout_ms)
{
    int ret;
    
    ret = bq27z746_mac_command(bq->client, cmd);
    if (ret)
        return ret;
    
    /* 根据命令类型等待 */
    msleep(timeout_ms);
    
    return 0;
}

/* 4. 通信失败重试 */
#define BQ27Z746_RETRY_COUNT    3
#define BQ27Z746_RETRY_DELAY_MS 10

static int bq27z746_read_with_retry(struct bq27z746_device *bq, u8 reg)
{
    int ret;
    int retry;
    
    for (retry = 0; retry < BQ27Z746_RETRY_COUNT; retry++) {
        ret = i2c_smbus_read_word_data(bq->client, reg);
        if (ret >= 0)
            return ret;
        
        msleep(BQ27Z746_RETRY_DELAY_MS);
    }
    
    dev_err(&bq->client->dev, "Read reg 0x%02x failed after %d retries\n",
            reg, BQ27Z746_RETRY_COUNT);
    return ret;
}
```

### 2. 配置更新模式注意事项

```c
 * CONFIG UPDATE 模式关键注意事项

/*
 * ⚠️ 警告: 在 CONFIG UPDATE 模式下:
 * 1. 电量计算暂停
 * 2. 保护功能可能受影响
 * 3. 不要长时间停留在此模式

static int bq27z746_safe_config_update(struct bq27z746_device *bq,
                                        int (*config_func)(struct bq27z746_device *))
{

    unsigned long start_time;
    
    /* 记录开始时间 */
    start_time = jiffies;

    ret = bq27z746_enter_config_update(bq);
    if (ret) {
        dev_err(&bq->client->dev, "Failed to enter config update\n");
        return ret;
    }
    
    /* 执行配置操作 */
    ret = config_func(bq);
    
    /* 无论成功与否，都要退出配置更新模式 */
    bq27z746_exit_config_update(bq);

                 jiffies_to_msecs(jiffies - start_time));
    }
    
    return ret;
}

 * 配置更新后需要等待电量计重新稳定
 * 建议等待 1-2 个测量周期 (约 1-2 秒)
 */
```

```c
{

        return ret;
```

```c
 * 问题: 产品进入 SHIP 模式后，在运输/存储过程中意外退出
 * 
 * 原因:
 * 1. 系统漏电流超过 IWAKE 阈值
 * 2. 发生复位事件
 * 3. 静电放电 (ESD) 触发
 */

/* 解决方案 1: 设置 SHIPM_ON_RST 位 */
/*
 * 当设置此位后，即使发生复位，设备也会重新进入 SHIP 模式
 */
static int enable_ship_mode_on_reset(struct bq27z746_device *bq)
{
    u16 power_config;
    int ret;
    
    ret = bq27z746_read_df_word(bq->client, DF_POWER_CONFIG, &power_config);
    if (ret)
        return ret;
    
    power_config |= PWRCFG_SHIPM_ON_RST;
    
    return bq27z746_write_df_word(bq->client, DF_POWER_CONFIG, power_config);
}

/* 解决方案 2: 调整 IWAKE 阈值 */
/*
 * 如果系统漏电流较大，可以提高 IWAKE 阈值
 * 或者禁用 IWAKE 功能
 */

/* 解决方案 3: 使用 SHELF 模式代替 SHIP 模式 */
/*
 * SHELF 模式不支持 IWAKE 唤醒，更不容易意外退出
 * 但只能通过充电电压唤醒
 */
```

```c
 * SOC 跳变问题分析
 */

 * 正常的 SOC 跳变:
 * - 在 OCV 测量点 (长时间静置后) 会有小幅校正
 * - 这是 Impedance Track 算法的正常行为
 * 
 * 异常的 SOC 跳变:
 * - 充电到 100% 时突然跳变
 * - 放电到 0% 时突然跳变
 * - 跳变幅度 > 5%
 */

/* 检查相关标志 */
static void check_soc_flags(struct bq27z746_device *bq)
{
    u16 battery_status;
    
    battery_status = bq27z746_read_word(bq->client, REG_BATTERY_STATUS);
    
    /* FC (Full Charge) 标志 */
    if (battery_status & FLAG_FC) {
        /* 已检测到满充，SOC 应该接近 100% */
    }
    
    /* FD (Full Discharge) 标志 */
    if (battery_status & FLAG_FD) {
        /* 已检测到完全放电，SOC 应该接近 0% */
    }
    
    /* TCA (Terminate Charge Alarm) */
    if (battery_status & FLAG_TCA) {
        /* 充电即将结束 */
    }
    
    /* TDA (Terminate Discharge Alarm) */
    if (battery_status & FLAG_TDA) {
        /* 放电即将结束 */
    }
}

/*
 * 减少 SOC 跳变的配置:
 * 1. 正确设置 Taper Current (涓流充电电流)
 * 2. 正确设置 Terminate Voltage (终止电压)
 * 3. 启用 SOC 平滑功能 (如果支持)
```

```c
/**
 * bq27z746_set_temp_protection - 设置温度保护参数
 */
static int bq27z746_set_temp_protection(struct bq27z746_device *bq,
                                         const temp_protection_config_t *cfg)
{
    int ret;

    ret = bq27z746_enter_config_update(bq);
    if (ret)
        return ret;

    /* OTC 配置 */

    bq27z746_write_df_word(bq->client, DF_OTC_RECOVERY, cfg->otc_recovery);
    bq27z746_write_df_word(bq->client, DF_OTC_DELAY, cfg->otc_delay);
```

```c
 * 状态转换逻辑 (以 OTC 为例):
 * 
 * NORMAL ──[T > OTC_Threshold]──> OTC_PENDING
 *                                      │
 *                                      │ [延迟时间到]
 *                                      ▼
 *                                 OTC_ACTIVE ──[关闭 CHG FET]
 *                                      │
 *                                      │ [T < OTC_Recovery]
 *                                      ▼
 *                                   NORMAL ──[开启 CHG FET]
 *

{
    u32 safety_status;
    
    bq27z746_mac_read_u32(bq->client, MAC_SAFETY_STATUS, &safety_status);
    
    if (safety_status & SAFETY_OTC) {
        dev_warn(&bq->client->dev, 
                 "OTC active - Charging disabled due to high temperature\n");
    }
    
    if (safety_status & SAFETY_OTD) {
        dev_warn(&bq->client->dev, 
                 "OTD active - Discharging disabled due to high temperature\n");
    }
    
    if (safety_status & SAFETY_UTC) {
        dev_warn(&bq->client->dev, 
                 "UTC active - Charging disabled due to low temperature\n");
    }
    
    if (safety_status & SAFETY_UTD) {
```

```c
    .hscd_delay     = 50,

/**
 * 计算电流阈值对应的 Rsense 电压
 * 用于验证配置是否在 ADC 测量范围内
 */
static int calculate_sense_voltage(int current_ma, int rsense_mohm)
{
    /* V = I × R */
    return (current_ma * rsense_mohm) / 1000;  /* 返回 mV */
}

/**
 * 验证电流保护配置
 */
static int validate_current_config(const current_protection_config_t *cfg,
                                    int rsense_mohm)
{
    int max_sense_voltage = 32;  /* BQ27Z746 最大检测电压 32.768mV */
    int sense_mv;
    
    /* 检查短路阈值是否超出测量范围 */
    sense_mv = calculate_sense_voltage(cfg->hscd_threshold, rsense_mohm);
    if (sense_mv > max_sense_voltage) {
        pr_warn("HSCD threshold %dmA exceeds measurement range "
                "(sense voltage %dmV > %dmV)\n",
                cfg->hscd_threshold, sense_mv, max_sense_voltage);
        return -EINVAL;
    }
    
    /* 检查保护层级是否合理 */
    if (cfg->ocd_threshold >= cfg->hocd_threshold) {
        pr_warn("OCD threshold should be less than HOCD threshold\n");
        return -EINVAL;
    }
    
    if (cfg->hocd_threshold >= cfg->hscd_threshold) {
        pr_warn("HOCD threshold should be less than HSCD threshold\n");
        return -EINVAL;
    }
    
    return 0;
}
```

### 1. 恢复机制类型

```c
/*
 * BQ27Z746 支持的保护恢复机制
 */

typedef enum {
    /* 自动恢复 - 条件满足后自动恢复 */
    RECOVERY_AUTO,
    
    /* 延迟恢复 - 条件满足后延迟一段时间恢复 */
    RECOVERY_DELAYED,
    
    /* 手动恢复 - 需要发送命令或重新上电 */
    RECOVERY_MANUAL,
    
    /* 锁存 - 永久锁定，需要特殊操作解除 */
    RECOVERY_LATCHED,
} recovery_type_t;

/*
 * 各保护类型的默认恢复机制:
 * 
 * 保护类型          恢复机制              恢复条件
 * ─────────────────────────────────────────────────────────
 * 固件过压 (COV)    自动恢复              电压 < 恢复阈值
 * 固件欠压 (CUV)    自动恢复              电压 > 恢复阈值
 * 固件过流 (OCC)    延迟自动恢复          电流正常 + 延迟时间
 * 固件过流 (OCD)    延迟自动恢复          电流正常 + 延迟时间
 * 固件过温          自动恢复              温度 < 恢复阈值
 * 固件低温          自动恢复              温度 > 恢复阈值
 * 硬件过压 (HCOV)   可配置 (自动/锁存)    取决于配置
 * 硬件欠压 (HCUV)   可配置 (自动/锁存)    取决于配置
 * 硬件过流          可配置 (自动/锁存)    取决于配置
 * 硬件短路 (HSCD)   通常锁存              需要移除负载或命令
 */
```

### 2. 恢复阈值设计

```c
/*
 * 恢复阈值设计原则
 * 
 * 迟滞 = 触发阈值 - 恢复阈值 (或反之)
 * 
 * 迟滞设计考虑:
 * 1. 太小: 可能在阈值附近频繁触发/恢复
 * 2. 太大: 恢复困难，影响用户体验
 */

typedef struct {
    /* 参数名称 */
    const char *name;
    
    /* 触发阈值 */
    int trigger;
    
    /* 恢复阈值 */
    int recovery;
    
    /* 迟滞值 */
    int hysteresis;
    
    /* 迟滞百分比 */
    int hysteresis_percent;
} threshold_design_t;

/* 推荐的迟滞设计 */
static const threshold_design_t recommended_hysteresis[] = {
    /* 电压保护 */
    { "COV",  4250, 4150, 100,  2 },   /* 100mV, ~2% */
    { "CUV",  2800, 3100, 300,  10 },  /* 300mV, ~10% */
    
    /* 温度保护 (单位: 0.1°C) */
    { "OTC",  450,  400,  50,   11 },  /* 5°C, ~11% */
    { "OTD",  550,  500,  50,   9 },   /* 5°C, ~9% */
    { "UTC",  0,    50,   50,   - },   /* 5°C */
    { "UTD",  -200, -100, 100,  - },   /* 10°C */
};

/*
 * 迟滞设计建议:
 * 
 * 电压:
 * - 过压迟滞: 50-100mV (约 1-2%)
 * - 欠压迟滞: 200-300mV (约 7-10%)
 *   欠压迟滞较大是因为低电压时电池内阻大，
 *   移除负载后电压回弹明显
 * 
 * 温度:
 * - 过温迟滞: 3-10°C
 * - 低温迟滞: 5-10°C
 *   温度变化较慢，需要较大迟滞避免频繁触发
 * 
 * 电流:
 * - 通常不设置恢复阈值
 * - 使用恢复延迟时间代替
 */
```

### 2. 配置验证函数

```c
/**
 * 验证保护配置的合理性
 */
static int validate_protection_config(const void *config)
{
    const typeof(smartphone_battery_config) *cfg = config;
    int errors = 0;
    
    /* 验证电压配置 */
    if (cfg->voltage.cov_threshold <= cfg->voltage.cov_recovery) {
        pr_err("COV threshold must be > recovery\n");
        errors++;
    }
    
    if (cfg->voltage.cuv_threshold >= cfg->voltage.cuv_recovery) {
        pr_err("CUV threshold must be < recovery\n");
        errors++;
    }
    
    if (cfg->voltage.hcov_threshold <= cfg->voltage.cov_threshold) {
        pr_err("HCOV threshold must be > COV threshold\n");
        errors++;
    }
    
    if (cfg->voltage.hcuv_threshold >= cfg->voltage.cuv_threshold) {
        pr_err("HCUV threshold must be < CUV threshold\n");
        errors++;
    }
    
    /* 验证电流配置 */
    if (cfg->current.hocd_threshold <= cfg->current.ocd_threshold) {
        pr_err("HOCD threshold must be > OCD threshold\n");
        errors++;
    }
    
    if (cfg->current.hscd_threshold <= cfg->current.hocd_threshold) {
        pr_err("HSCD threshold must be > HOCD threshold\n");
        errors++;
    }
    
    /* 验证温度配置 */
    if (cfg->temp.otc_threshold <= cfg->temp.otc_recovery) {
        pr_err("OTC threshold must be > recovery\n");
        errors++;
    }
    
    if (cfg->temp.utc_threshold >= cfg->temp.utc_recovery) {
        pr_err("UTC threshold must be < recovery\n");
        errors++;
    }
    
    /* 验证迟滞合理性 */
    int cov_hysteresis = cfg->voltage.cov_threshold - cfg->voltage.cov_recovery;
    if (cov_hysteresis < 50 || cov_hysteresis > 200) {
        pr_warn("COV hysteresis %dmV may not be optimal (recommend 50-200mV)\n",
                cov_hysteresis);
    }
    
    int otc_hysteresis = cfg->temp.otc_threshold - cfg->temp.otc_recovery;
    if (otc_hysteresis < 30 || otc_hysteresis > 100) {
        pr_warn("OTC hysteresis %d (0.1°C) may not be optimal\n",
                otc_hysteresis);
    }
    
    return errors ? -EINVAL : 0;
}
```

### 3. 配置验证函数

```c
/**
 * validate_protection_config_full - 完整验证保护配置
 */
static int validate_protection_config_full(const bq27z746_protection_config_t *cfg)
{
    int errors = 0;
    
    /* === 电压配置验证 === */
    
    /* COV: 触发 > 恢复 */
    if (cfg->voltage.cov_threshold_mv <= cfg->voltage.cov_recovery_mv) {
        pr_err("COV: threshold (%d) must be > recovery (%d)\n",
               cfg->voltage.cov_threshold_mv, cfg->voltage.cov_recovery_mv);
        errors++;
    }
    
    /* COV 迟滞检查 (推荐 50-150mV) */
    int cov_hyst = cfg->voltage.cov_threshold_mv - cfg->voltage.cov_recovery_mv;
    if (cov_hyst < 50 || cov_hyst > 200) {
        pr_warn("COV hysteresis %dmV outside recommended range (50-200mV)\n", cov_hyst);
    }
    
    /* CUV: 触发 < 恢复 */
    if (cfg->voltage.cuv_threshold_mv >= cfg->voltage.cuv_recovery_mv) {
        pr_err("CUV: threshold (%d) must be < recovery (%d)\n",
               cfg->voltage.cuv_threshold_mv, cfg->voltage.cuv_recovery_mv);
        errors++;
    }
    
    /* CUV 迟滞检查 (推荐 200-400mV) */
    int cuv_hyst = cfg->voltage.cuv_recovery_mv - cfg->voltage.cuv_threshold_mv;
    if (cuv_hyst < 150 || cuv_hyst > 500) {
        pr_warn("CUV hysteresis %dmV outside recommended range (150-500mV)\n", cuv_hyst);
    }
    
    /* 硬件保护 > 固件保护 */
    if (cfg->voltage.hcov_threshold_mv <= cfg->voltage.cov_threshold_mv) {
        pr_err("HCOV (%d) must be > COV (%d)\n",
               cfg->voltage.hcov_threshold_mv, cfg->voltage.cov_threshold_mv);
        errors++;
    }
    
    if (cfg->voltage.hcuv_threshold_mv >= cfg->voltage.cuv_threshold_mv) {
        pr_err("HCUV (%d) must be < CUV (%d)\n",
               cfg->voltage.hcuv_threshold_mv, cfg->voltage.cuv_threshold_mv);
        errors++;
    }
    
    /* === 电流配置验证 === */
    
    /* 保护层级: OCC < HOCC */
    if (cfg->current.hocc_threshold_ma <= cfg->current.occ_threshold_ma) {
        pr_err("HOCC (%d) must be > OCC (%d)\n",
               cfg->current.hocc_threshold_ma, cfg->current.occ_threshold_ma);
        errors++;
    }
    
    /* 保护层级: OCD < HOCD < HSCD */
    if (cfg->current.hocd_threshold_ma <= cfg->current.ocd_threshold_ma) {
        pr_err("HOCD (%d) must be > OCD (%d)\n",
               cfg->current.hocd_threshold_ma, cfg->current.ocd_threshold_ma);
        errors++;
    }
    
    if (cfg->current.hscd_threshold_ma <= cfg->current.hocd_threshold_ma) {
        pr_err("HSCD (%d) must be > HOCD (%d)\n",
               cfg->current.hscd_threshold_ma, cfg->current.hocd_threshold_ma);
        errors++;
    }
    
    /* === 温度配置验证 === */
    
    /* OTC: 触发 > 恢复 */
    if (cfg->temperature.otc_threshold <= cfg->temperature.otc_recovery) {
        pr_err("OTC: threshold (%d) must be > recovery (%d)\n",
               cfg->temperature.otc_threshold, cfg->temperature.otc_recovery);
        errors++;
    }
    
    /* OTC 迟滞检查 (推荐 30-100, 即 3-10°C) */
    int otc_hyst = cfg->temperature.otc_threshold - cfg->temperature.otc_recovery;
    if (otc_hyst < 30 || otc_hyst > 100) {
        pr_warn("OTC hysteresis %d (0.1°C) outside recommended range\n", otc_hyst);
    }
    
    /* OTD: 触发 > 恢复 */
    if (cfg->temperature.otd_threshold <= cfg->temperature.otd_recovery) {
        pr_err("OTD: threshold (%d) must be > recovery (%d)\n",
               cfg->temperature.otd_threshold, cfg->temperature.otd_recovery);
        errors++;
    }
    
    /* UTC: 触发 < 恢复 */
    if (cfg->temperature.utc_threshold >= cfg->temperature.utc_recovery) {
        pr_err("UTC: threshold (%d) must be < recovery (%d)\n",
               cfg->temperature.utc_threshold, cfg->temperature.utc_recovery);
        errors++;
    }
    
    /* UTD: 触发 < 恢复 */
    if (cfg->temperature.utd_threshold >= cfg->temperature.utd_recovery) {
        pr_err("UTD: threshold (%d) must be < recovery (%d)\n",
               cfg->temperature.utd_threshold, cfg->temperature.utd_recovery);
        errors++;
    }
    
    /* OTC 应该 < OTD (充电温度限制更严格) */
    if (cfg->temperature.otc_threshold >= cfg->temperature.otd_threshold) {
        pr_warn("OTC threshold (%d) should typically be < OTD threshold (%d)\n",
                cfg->temperature.otc_threshold, cfg->temperature.otd_threshold);
    }
    
    /* UTC 应该 > UTD (充电温度限制更严格) */
    if (cfg->temperature.utc_threshold <= cfg->temperature.utd_threshold) {
        pr_warn("UTC threshold (%d) should typically be > UTD threshold (%d)\n",
                cfg->temperature.utc_threshold, cfg->temperature.utd_threshold);
    }
    
    return errors ? -EINVAL : 0;
}
```

### 阈值设计核心原则

```text
┌─────────────────────────────────────────────────────────────────────────────────┐
│                    保护阈值设计核心原则总结                                     │
├─────────────────────────────────────────────────────────────────────────────────┤
│                                                                                 │
│  1. 层级原则                                                                    │
│     正常工作范围 < 固件保护 < 硬件保护 < 电池物理极限                           │
│                                                                                 │
│  2. 迟滞原则                                                                    │
│     触发阈值和恢复阈值之间必须有足够的迟滞，避免抖动                            │
│     - 电压: 50-300mV                                                            │
│     - 温度: 3-10°C                                                              │
│                                                                                 │
│  3. 延迟原则                                                                    │
│     - 短路保护: 最快 (µs 级)                                                    │
│     - 过流保护: 较快 (ms 级)                                                    │
│     - 电压/温度保护: 较慢 (s 级)                                                │
│                                                                                 │
│  4. 裕量原则                                                                    │
│     - 固件保护 = 正常工作极限 × 1.1~1.2                                         │
│     - 硬件保护 = 固件保护 × 1.3~1.5                                             │
│                                                                                 │
│  5. 充放电差异原则                                                              │
│     - 充电保护通常比放电更严格                                                  │
│     - 充电温度范围比放电窄                                                      │
│     - 充电电流限制比放电低                                                      │
│                                                                                 │
│  6. 验证原则                                                                    │
│     - 必须进行功能测试验证                                                      │
│     - 必须进行边界条件测试                                                      │
│     - 必须进行多样品验证                                                        │
│                                                                                 │
└─────────────────────────────────────────────────────────────────────────────────┘
```
