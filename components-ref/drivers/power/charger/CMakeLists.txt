# XinYi Charger Driver - BQ25620
# TI BQ25620 Standalone I2C 1-Cell Li-Ion Battery Charger Driver

set(CHARGER_SOURCES
    ${CMAKE_CURRENT_LIST_DIR}/xy_bq25620.c
)

set(CHARGER_HEADERS
    ${CMAKE_CURRENT_LIST_DIR}/xy_bq25620.h
)

# 添加到父级变量
list(APPEND COMPONENT_SOURCES ${CHARGER_SOURCES})
list(APPEND COMPONENT_HEADERS ${CHARGER_HEADERS})

# 添加包含目录
list(APPEND COMPONENT_INCLUDE_DIRS ${CMAKE_CURRENT_LIST_DIR})

# 设置父级变量
set(COMPONENT_SOURCES ${COMPONENT_SOURCES} PARENT_SCOPE)
set(COMPONENT_HEADERS ${COMPONENT_HEADERS} PARENT_SCOPE)
set(COMPONENT_INCLUDE_DIRS ${COMPONENT_INCLUDE_DIRS} PARENT_SCOPE)
